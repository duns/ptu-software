setenv bootdelay '5'
setenv bootscript 'echo Running bootscript from mmc ...; source ${loadaddr}'
setenv console 'ttyO2,115200n8'
setenv defaultdisplay 'dvi'
setenv dieid# '626e00029ff800000168300f0500e01b'
setenv dvimode '1024x768MR-16@60'
setenv loadaddr '0x82000000'
setenv loadbootscript 'fatload mmc ${mmcdev} ${loadaddr} boot.scr'
setenv loaduimage 'fatload mmc ${mmcdev} ${loadaddr} uImage'
setenv mmcargs 'setenv bootargs console=${console} ${optargs} mpurate=${mpurate} vram=${vram} omapfb.mode=dvi:${dvimode} omapdss.def_disp=${defaultdisplay} root=${mmcroot} rootfstype=${mmcrootfstype}'
setenv mmcboot 'echo Booting from mmc ...; run mmcargs; bootm ${loadaddr}'
setenv mmcdev '0'
setenv mmcroot '/dev/mmcblk0p2 rw'
setenv mmcrootfstype 'ext3 rootwait'
setenv mpurate '720'
setenv nandargs 'setenv bootargs console=${console} ${optargs} mpurate=${mpurate} vram=${vram} omapfb.mode=dvi:${dvimode} omapdss.def_disp=${defaultdisplay} root=${nandroot} rootfstype=${nandrootfstype}'
setenv nandboot 'echo Booting from nand ...; run nandargs; nand read ${loadaddr} 280000 400000; bootm ${loadaddr}'
setenv nandroot 'ubi0:rootfs ubi.mtd=4'
setenv nandrootfstype 'ubifs'
setenv optargs 'mem=99M@0x80000000 mem=128M@0x88000000'
setenv stderr 'serial'
setenv stdin 'serial'
setenv stdout 'serial'
